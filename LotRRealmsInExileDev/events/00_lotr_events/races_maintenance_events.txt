lotrracesmaint.001 = { # test1
	type = character_event
	hidden = yes
	
	trigger = {
		AND = {
			is_alive = yes
			NOT = { has_character_flag = fixed_species }
			}
	}

	weight_multiplier = {
		base = 1
	}

	#
	immediate = {
		add_character_flag = fixed_species
		if = {
			limit = {
					AND={
						OR={
						has_culture_group  = lotr_amanyar_group
						has_culture_group  = lotr_umanyar_group
						has_culture_group  = lotr_moriquendi_group
						has_culture_group  = lotr_quendi_group
						}
						NOT={
					has_trait = elf
						}
					}
				}
				add_trait = elf
		}
			
			#
		if = {
			limit = {
					AND={
					has_culture_group  = lotr_ainur_group	
					NOT={
						has_trait = valar
						}
					}
				add_trait = valar
			}
				
		}

	}
}
